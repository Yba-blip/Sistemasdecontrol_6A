    <form id="trianguloForm">
        <label for="base">Base:</label>
        <input type="number" id="base" step="any" required>
        <br>
        <label for="altura">Altura:</label>
        <input type="number" id="altura" step="any" required>
        <br>
        <label for="lado1">Lado 1:</label>
        <input type="number" id="lado1" step="any" required>
        <br>
        <label for="lado2">Lado 2:</label>
        <input type="number" id="lado2" step="any" required>
        <br>
        <label for="lado3">Lado 3:</label>
        <input type="number" id="lado3" step="any" required>
        <br>
        <button type="button" onclick="calcularTriangulo()">Calcular</button>
    </form>

    <div id="resultado">
    </div>

    <script>
        function calcularTriangulo() {
            const base = parseFloat(document.getElementById("base").value);
            const altura = parseFloat(document.getElementById("altura").value);
            const lado1 = parseFloat(document.getElementById("lado1").value);
            const lado2 = parseFloat(document.getElementById("lado2").value);
            const lado3 = parseFloat(document.getElementById("lado3").value);

            // Función para calcular la hipotenusa
            function hipotenusa(base, altura) {
                return Math.sqrt(base**2 + altura**2);
            }

            // Función para calcular el perímetro
            function calcularPerimetro(base, altura, hip) {
                return base + altura + hip;
            }

            // Función para calcular la superficie
            function calcularSuperficie(base, altura) {
                return 0.5 * base * altura;
            }

            // Función para verificar si tres lados forman un triángulo
            function esTriangulo(a, b, c) {
                return (a + b > c) && (a + c > b) && (b + c > a);
            }

            let resultado = document.getElementById("resultado");
            resultado.innerHTML = "";

            if (esTriangulo(lado1, lado2, lado3)) {
                const hip = hipotenusa(base, altura);
                const perimetro = calcularPerimetro(base, altura, hip);
                const superficie = calcularSuperficie(base, altura);
                resultado.innerHTML = `
                    <p>Hipotenusa: ${hip}</p>
                    <p>Perímetro: ${perimetro}</p>
                    <p>Superficie: ${superficie}</p>
                    <p>Los lados ingresados forman un triángulo.</p>
                `;
            } else {
                resultado.innerHTML = `<p>Los lados ingresados no forman un triángulo.</p>`;
            }
        }
    </script>
